% Palette: calm blue-gray, not harsh on screen/print
\definecolor{cvprimary}{HTML}{2C5282}
\definecolor{cvprimarysoft}{HTML}{4A6FA5}
\definecolor{cvheaderbg}{HTML}{F0F4FA}
\definecolor{cvruled}{HTML}{C8D4E8}
\definecolor{cvmuted}{HTML}{5A6570}

\hypersetup{
  colorlinks=true,
  linkcolor=cvprimary,
  urlcolor=cvprimarysoft,
  pdftitle={Pierre Courteille - CV},
  pdfauthor={Pierre Courteille}
}

% Section titles: centered, color only (no bold) + soft rule
\titleformat{\section}
  {\centering\large\color{cvprimary}}
  {}
  {0pt}
  {}
  [\vspace{0.25em}{\color{cvruled}\rule{\linewidth}{0.55pt}}]

\titlespacing*{\section}{0pt}{1.05em plus 0.15em}{0.55em plus 0.08em}

% Default lists (profile, certs, education)
\setlist[itemize]{
  noitemsep,
  topsep=0.35em,
  itemsep=0.2em,
  leftmargin=1.2em,
  label=\textcolor{cvprimary}{\textbullet},
  before={\RaggedRight}
}

% Softer bullets for dense experience blocks
\newlist{cvitemize}{itemize}{1}
\setlist[cvitemize]{
  label=\textcolor{cvprimarysoft}{\small$\triangleright$},
  leftmargin=1.15em,
  itemsep=0.22em,
  topsep=0.25em,
  parsep=0pt,
  partopsep=0pt,
  before={\RaggedRight}
}

% Job title: colored, slightly larger — not bold (lighter than body bold spam)
\newcommand{\cvSubheading}[3]{%
  \par\vspace{0.65em}%
  {\large\color{cvprimary}#1}\par\vspace{0.1em}
  {\small\color{cvmuted}#3 \quad\textbar\quad \textit{#2}}\par
  \vspace{0.3em}
}

\newcommand{\cvStack}[1]{%
  \par\noindent{\small\color{cvmuted}\textit{Stack:}\ #1}\par
}

% Bullet lead label: accent color, regular weight (replaces heavy \textbf)
\newcommand{\cvTopic}[1]{\textcolor{cvprimary}{#1}}
