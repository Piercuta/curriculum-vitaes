% Palette: darker blue for stronger contrast (titles, links, accents)
\definecolor{cvprimary}{HTML}{153E75}
\definecolor{cvprimarysoft}{HTML}{1E4976}
\definecolor{cvheaderbg}{HTML}{E8EEF7}
\definecolor{cvruled}{HTML}{9BB3D1}
\definecolor{cvmuted}{HTML}{5A6570}

\hypersetup{
  colorlinks=true,
  linkcolor=cvprimary,
  urlcolor=cvprimarysoft,
  pdftitle={Pierre Courteille - CV},
  pdfauthor={Pierre Courteille}
}

% Section titles: centered, bold, slightly larger + rule below
\titleformat{\section}
  {\centering\Large\bfseries\color{cvprimary}}
  {}
  {0pt}
  {}
  [\vspace{0.28em}{\color{cvruled}\rule{\linewidth}{0.6pt}}]

\titlespacing*{\section}{0pt}{0.85em plus 0.12em}{0.45em plus 0.06em}

% Default lists (profile, certs, education)
\setlist[itemize]{
  noitemsep,
  topsep=0.35em,
  itemsep=0.2em,
  leftmargin=1.2em,
  label=\textcolor{cvprimary}{\textbullet},
  before={\RaggedRight}
}

% Softer bullets for dense experience blocks
\newlist{cvitemize}{itemize}{1}
\setlist[cvitemize]{
  label=\textcolor{cvprimarysoft}{\small$\triangleright$},
  leftmargin=1.15em,
  itemsep=0.22em,
  topsep=0.25em,
  parsep=0pt,
  partopsep=0pt,
  before={\RaggedRight}
}

% Job title: colored, slightly larger — not bold (lighter than body bold spam)
\newcommand{\cvSubheading}[3]{%
  \par\vspace{0.45em}%
  {\large\color{cvprimary}#1}\par\vspace{0.08em}
  {\small\color{cvmuted}#3 \quad\textbar\quad \textit{#2}}\par
  \vspace{0.22em}
}

\newcommand{\cvStack}[1]{%
  \par\noindent{\small\color{cvmuted}\textit{Stack:}\ #1}\par
}

% Visible break between experience blocks (after each job except you omit after the last)
\newcommand{\cvJobSeparator}{%
  \par\vspace{0.55em}
  {\color{cvruled}\rule{\linewidth}{0.65pt}}%
  \par\vspace{0.55em}
}

% Bullet lead label: accent color, regular weight (replaces heavy \textbf)
\newcommand{\cvTopic}[1]{\textcolor{cvprimary}{#1}}
